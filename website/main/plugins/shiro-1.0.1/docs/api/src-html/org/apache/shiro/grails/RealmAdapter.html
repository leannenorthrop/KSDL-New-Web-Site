021    package org.apache.shiro.grails;
022    
023    import org.apache.shiro.realm.Realm;
024    import org.apache.shiro.subject.PrincipalCollection;
025    import org.apache.shiro.authz.AuthorizationException;
026    import org.apache.shiro.authz.Permission;
027    import org.apache.shiro.authz.permission.InvalidPermissionStringException;
028    import org.apache.shiro.authz.permission.PermissionResolver;
029    import org.apache.shiro.authz.permission.PermissionResolverAware;
030    import org.apache.shiro.authz.permission.WildcardPermission;
031    
032    import java.util.List;
033    import java.util.ArrayList;
034    
035    
036    /**
037     * Adapter for the RealmWrapper which has problems implementing the
038     * "String..." methods under JDK 1.5. So, these methods automatically
039     * convert the strings into WildcardPermission instances.
040     */
041    public abstract class RealmAdapter implements Realm, PermissionResolverAware {
042        PermissionResolver permResolver = null;
043    
044        public void setPermissionResolver(PermissionResolver pr) {
045            this.permResolver = pr;
046        }
047    
048        public void checkPermission(PrincipalCollection principal, String s) throws AuthorizationException {
049            checkPermission(principal, toPermission(s));
050        }
051    
052        public void checkPermissions(PrincipalCollection principal, String... strings) throws AuthorizationException {
053            checkPermissions(principal, toPermissionList(strings));
054        }
055    
056        public boolean isPermitted(PrincipalCollection principal, String s) {
057            return isPermitted(principal, toPermission(s));
058        }
059    
060        public boolean[] isPermitted(PrincipalCollection principal, String... strings) {
061            return isPermitted(principal, toPermissionList(strings));
062        }
063    
064        public boolean isPermittedAll(PrincipalCollection principal, String... strings) {
065            return isPermittedAll(principal, toPermissionList(strings));
066        }
067    
068        /**
069         * Converts a single permission string into a Permission instances.
070         */
071        private Permission toPermission(String s) {
072            if (permResolver == null) return null;
073            try {
074                return permResolver.resolvePermission(s);
075            }
076            catch (InvalidPermissionStringException ex) {
077                // Nothing we can do about it
078                return null; //@todo Is returning null the right thing to do?
079            }
080        }
081    
082        /**
083         * Converts an array of string permissions into a list of
084         * {@link WildcardPermission} instances.
085         */
086        private List toPermissionList(String[] strings) {
087            List permissions = new ArrayList(strings.length);
088            for (int i = 0; i < strings.length; i++) {
089                permissions.add(toPermission(strings[i]));
090            }
091    
092            return permissions;
093        }
094    }
